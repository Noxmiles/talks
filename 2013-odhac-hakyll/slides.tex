\begin{frame}[fragile]{Hakyll}
    XMonad-like configuration: \\
    \vspaced
    \begin{lstlisting}
    match "*.markdown" $ do
      route   $ setExtension "html"
      compile $ pandocCompiler
        >>= relativizeUrls
    \end{lstlisting}
