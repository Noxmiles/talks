\end{frame}

\begin{frame}[fragile]{Composability}
    \begin{lstlisting}

    \end{lstlisting}
\end{frame}

\begin{frame}[fragile]{Composability}
    \begin{lstlisting}

    \end{lstlisting}
\end{frame}

\begin{frame}[fragile]{Composability}
    \begin{lstlisting}
    any f list = or (map f list)
    \end{lstlisting}

\end{frame}

\begin{frame}[fragile]{Composability}
