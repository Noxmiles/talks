\begin{frame}[fragile]{Example: FRP}
    \begin{lstlisting}
    data Signal a

    instance Functor Signal
    instance Monad Signal
    instance Applicative Signal
    \end{lstlisting}
\end{frame}

\begin{frame}[fragile]{Example: FRP}
    \begin{lstlisting}
    dollars :: Signal Int
    dollars = do
      instrs <- portfolio
      return $ sum $
        map instrumentValue instrs
    \end{lstlisting}
\end{frame}
